commands:
  kmcprefs:
    description: View or manage player game preferences
    usage: /kmcprefs <view|save|reset|cleargame> [args]
    permission: kmc.preferences.admin

  kmchealth:
    description: View KMCCore health monitor events
    usage: /kmchealth [recent|warnings|critical]
    permission: kmc.health.admin

  kmcready:
    description: Admin override for ready-up checks
    usage: /kmcready <force|status>
    permission: kmc.ready.admin

  kmcmap:
    description: Manage map rotation per game
    usage: /kmcmap <list|add|remove|mode|setfixed|pick> [args]
    permission: kmc.map.admin

  kmclobbynpc:
    description: Spawn lobby stat NPCs
    usage: /kmclobbynpc <spawn|despawnall> [type]
    permission: kmc.lobbynpc.admin

permissions:
  kmc.preferences.admin:
    description: Manage player preferences
    default: op
  kmc.health.admin:
    description: View health events
    default: op
  kmc.ready.admin:
    description: Override ready checks
    default: op
  kmc.map.admin:
    description: Manage map rotation
    default: op
  kmc.lobbynpc.admin:
    description: Spawn/despawn lobby NPCs
    default: op
